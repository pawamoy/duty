64 def format_help(self) -> str: 

65 """ 

66 Format a message listing the duties. 

67 

68 Returns: 

69 A string listing the duties and their summary. 

70 """ 

71 lines = [] 

72 # 20 makes the summary aligned with options description 

73 longest_name = max(max(len(name) for name in self.duties), 20) # noqa: WPS432 (magic number) 

74 for name, duty in self.duties.items(): 

75 description = duty.description.split("\n")[0] 

76 lines.append(f"{name:{longest_name}} {description}") 

77 return "\n".join(lines)

80 """

85 Uses the collection's path by default.

87 path = path or self.path 

88 spec = importlib_util.spec_from_file_location("duty.duties", path) 

89 duties = importlib_util.module_from_spec(spec) 

90 spec.loader.exec_module(duties) # type: ignore 

91 declared_duties = inspect.getmembers(duties, lambda member: isinstance(member, Duty)) 

92 for _, duty in declared_duties:

189 def run_duties(self, context, duties_list: DutyListType) -> None: # noqa: WPS231 (not complex) 

190 """ 

191 Run a list of duties. 

192 

193 Arguments: 

194 context: The context to use. 

195 duties_list: The list of duties to run. 

196 

197 Raises: 

198 RuntimeError: When a duty name is given to pre or post duties. 

199 Indeed, without a parent collection, it is impossible 

200 to find another duty by its name. 

201 """ 

202 for duty_item in duties_list: 

203 if callable(duty_item): 

204 # Item is a proper duty, or a callable: run it. 

205 duty_item(context) 

206 elif isinstance(duty_item, str): 

207 # Item is a reference to a duty. 

208 if self.collection is None: 

209 raise RuntimeError( 

210 f"Can't find duty by name without a collection ({duty_item})", 

211 ) 

212 # Get the duty and run it. 

213 self.collection.get(duty_item)(context)
