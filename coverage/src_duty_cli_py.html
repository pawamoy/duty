1# Why does this file exist, and why not put this in `__main__`? 

2# 

3# You might be tempted to import things from `__main__` later, 

4# but that will cause problems: the code will get executed twice: 

5# 

6# - When you run `python -m duty` python will execute 

7# `__main__.py` as a script. That means there won't be any 

8# `duty.__main__` in `sys.modules`. 

9# - When you import `__main__` it will get executed again (as a module) because 

10# there's no `duty.__main__` in `sys.modules`.

70def split_args(args: List[str], names: List[str]) -> List[List[str]]: # noqa: WPS231 (complex) 

71 """ 

72 Split command line arguments into duty commands. 

73 

74 Arguments: 

75 args: The CLI arguments. 

76 names: The known duty names. 

77 

78 Raises: 

79 ValueError: When a duty name is missing before an argument, 

80 or when the duty name is unknown. 

81 

82 Returns: 

83 The split commands. 

84 """ 

85 arg_lists = [] 

86 current_arg_list: List[str] = [] 

87 

88 for arg in args: 

89 if arg in names: 

90 # We found a duty name. 

91 if current_arg_list: 

92 # Append the previous arg list to the result and reset it. 

93 arg_lists.append(current_arg_list) 

94 current_arg_list = [] 

95 current_arg_list.append(arg) 

96 elif current_arg_list: 

97 # We found an argument. 

98 current_arg_list.append(arg) 

99 else: 

100 # We found an argument but no duty name. 

101 raise ValueError(f"> Missing duty name before argument '{arg}', or unknown duty name") 

102 

103 # Don't forget the last arg list. 

104 if current_arg_list: 

105 arg_lists.append(current_arg_list) 

106 

107 return arg_lists

130def specified_options(opts: argparse.Namespace, exclude=None) -> Dict: 

131 """ 

132 Cast an argparse Namespace into a dictionary of options. 

133 

134 Remove all options that were not specified (equal to None). 

135 

136 Arguments: 

137 opts: The namespace to cast. 

138 exclude: Names of options to exclude from the result. 

139 

140 Returns: 

141 A dictionary of specified-only options. 

142 """ 

143 exclude = exclude or set() 

144 options = opts.__dict__.items() # noqa: WPS609 

145 return {opt: value for opt, value in options if value is not None and opt not in exclude} # noqa: WPS221

148def parse_options(duty: Duty, args: List[str]) -> Tuple[Dict, List[str]]: 

149 """ 

150 Parse options for a duty. 

151 

152 Arguments: 

153 duty: The duty to parse for. 

154 args: The CLI args passed for this duty. 

155 

156 Returns: 

157 The parsed opts, and the remaining arguments. 

158 """ 

159 parser = get_duty_parser(duty) 

160 opts, remainder = parser.parse_known_args(args) 

161 return specified_options(opts), remainder

164def parse_args(duty: Duty, args: List[str]) -> Tuple: # noqa: WPS231 (complex) 

165 """ 

166 Parse the positional and keyword arguments of a duty. 

167 

168 Arguments: 

169 duty: The duty to parse for. 

170 args: The list of arguments. 

171 

172 Returns: 

173 The positional and keyword arguments. 

174 """ 

175 posargs = [] 

176 kwargs = {} 

177 

178 for arg in args: 

179 if "=" in arg: 

180 # we found a keyword argument 

181 arg_name, arg_value = arg.split("=", 1) 

182 kwargs[arg_name] = arg_value 

183 else: 

184 # we found a positional argument 

185 posargs.append(arg) 

186 

187 return validate(duty.function, *posargs, **kwargs)

190def parse_commands(arg_lists, global_opts, collection) -> List[Tuple]: 

191 """ 

192 Parse argument lists into ready-to-run duties. 

193 

194 Arguments: 

195 arg_lists: Lists of arguments lists. 

196 global_opts: The global options. 

197 collection: The duties collection. 

198 

199 Returns: 

200 A list of tuples composed of: 

201 

202 - a duty 

203 - its positional arguments 

204 - its keyword arguments 

205 """ 

206 commands = [] 

207 for arg_list in arg_lists: 

208 duty = collection.get(arg_list[0]) 

209 opts, remainder = parse_options(duty, arg_list[1:]) 

210 duty.options_override = {**global_opts, **opts} 

211 commands.append((duty, *parse_args(duty, remainder))) 

212 return commands

215def print_help(parser: ArgParser, opts: argparse.Namespace, collection: Collection) -> None: 

216 """ 

217 Print general help or duties help. 

218 

219 Arguments: 

220 parser: The main parser. 

221 opts: The main parsed options. 

222 collection: A collection of duties. 

223 """ 

224 if opts.help: 

225 for duty_name in opts.help: 

226 try: 

227 duty = collection.get(duty_name) 

228 except KeyError: 

229 print(f"> Unknown duty '{duty_name}'") 

230 else: 

231 print(get_duty_parser(duty).format_help()) 

232 else: 

233 print(parser.format_help()) 

234 print("Available duties:") 

235 print(textwrap.indent(collection.format_help(), prefix=" "))

238def main(args: Optional[List[str]] = None) -> int: # noqa: WPS212 (return statements) 

239 """ 

240 Run the main program. 

241 

242 This function is executed when you type `duty` or `python -m duty`. 

243 

244 Arguments: 

245 args: Arguments passed from the command line. 

246 

247 Returns: 

248 An exit code. 

249 """ 

250 parser = get_parser() 

251 opts = parser.parse_args(args=args) 

252 remainder = opts.remainder 

253 

254 collection = Collection(opts.duties_file) 

255 collection.load() 

256 

257 if opts.help is not None: 

258 print_help(parser, opts, collection) 

259 return 0 

260 

261 if opts.list: 

262 print(textwrap.indent(collection.format_help(), prefix=" ")) 

263 return 0 

264 

265 try: 

266 arg_lists = split_args(remainder, collection.names()) 

267 except ValueError as error: 

268 print(error, file=sys.stderr) 

269 return 1 

270 

271 if not arg_lists: 

272 print("> Please choose at least one duty", file=sys.stderr) 

273 return 1 

274 

275 global_opts = specified_options(opts, exclude={"duties_file", "list", "help", "remainder"}) 

276 try: 

277 commands = parse_commands(arg_lists, global_opts, collection) 

278 except TypeError as error: # noqa: WPS440 (variable overlap) 

279 print(f"> {error}", file=sys.stderr) 

280 return 1 

281 

282 for duty, posargs, kwargs in commands: 

283 try: 

284 duty.run(*posargs, **kwargs) 

285 except DutyFailure as failure: 

286 return failure.code 

287 

288 return 0
